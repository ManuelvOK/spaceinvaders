#include <stdio.h>
#include <ncurses.h>

int main() {
    WINDOW *menu_win;

    initscr();
    noecho();
    cbreak();   /* Line buffering disabled. pass on everything */
        
    menu_win = newwin(24, 80, 0, 0);
    keypad(menu_win, TRUE);
    wtimeout(menu_win,0);
    int c;
    while (1) {
        //c = getch();
        c = wgetch(menu_win);
        if (c != -1) {
            printf("%i\r\n",c);
        }
    }
    endwin();
    return 0;
}
